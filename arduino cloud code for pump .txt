#include <ArduinoIoTCloud.h>
#include <Arduino_ConnectionHandler.h>

const char DEVICE_LOGIN_NAME[] = "1111cd8b-7f5f-44fa-8cbb-d621335e04ff";

const char SSID[] = "vivo";
const char PASS[] = "123456789";
const char DEVICE_KEY[] = "GdFEXJxCmvDW89bjayDfpNB#i";
void onPumpChange();

bool pump;

void initProperties() {
  ArduinoCloud.setBoardId(DEVICE_LOGIN_NAME);
  ArduinoCloud.setSecretDeviceKey(DEVICE_KEY);
  ArduinoCloud.addProperty(pump, READWRITE, ON_CHANGE, onPumpChange);
}

WiFiConnectionHandler ArduinoIoTPreferredConnection(SSID, PASS);

void setup() {
  Serial.begin(115200);
  delay(2000);

  Serial.println("Starting...");

  initProperties();

  ArduinoCloud.begin(ArduinoIoTPreferredConnection);

  setDebugMessageLevel(4);
  ArduinoCloud.printDebugInfo();
}

void loop() {
  ArduinoCloud.update();

  static unsigned long lastPrint = 0;

  if (millis() - lastPrint > 5000) {
    lastPrint = millis();
    Serial.println("Cloud Running...");
  }
}
